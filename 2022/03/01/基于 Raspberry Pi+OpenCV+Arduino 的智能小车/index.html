基于 Raspberry Pi+OpenCV+Arduino 的智能小车

本项目基于上位机Raspberry Pi和下位机Arduino相互通信的环境，利用SSH远程登录，结合摄像头OpenCV和
Python进行图像的传输并判别交通信号的箭头，小车识别到绿色箭头朝箭头指示方向转弯，识别到红色箭头则
停在原地，并计算左右两侧的 PWM 值实现直线行驶，利用超声波模块检测小车前方障碍物距离实现避障。

设计方法

    本项目有两个控制板块，树莓派作为上位机，Arduino作为下位机，超声波模块用于检测前方一定距离内是否有箭头标志阻挡前进的路线，电机驱动模块以PWM方式控制左右两侧的轮子（左边两个轮子相同速度信号，右边两个轮子相同速度信号），通过左右车轮的速度差来达到转弯的效果。

    树莓派作为上位机，通过摄像头判别交通箭头的信号。判别箭头是通过判断识别到的轮廓（箭头轮廓）的重心和最小外接矩形的重心两者的位置，箭头的朝向跟箭头重心的位置直接关联，所以可以根据两个重心的位置关系来判断箭头是左箭头还是右箭头。关于颜色的识别，将RGB颜色转化位HSV空间的颜色，判断图片中的颜色是否在绿色或者红色的色域中，在色域中则说明是对应的颜色。

制作方案

环境的安装

主要涉及关于树莓派和OpenCV的安装
·1 树莓派安装（便于可视化可以配置一台显示屏，但意义就没有了，因此获取树莓派IP后可以通过SSH远程控制树莓派）
1.1 下载系统镜像（之前写过关于Ubuntu16.04下Git的安装）,此处使用的是Raspbian的镜像（自带桌面和一些软件）
1.2 格式化SD卡，将镜像写入，完成系统的安装
1.3 开启SSH
首先在根目录下新建SSH文件，用于远程登录，然后就把SD卡插到树莓派上；
第一次需要用网线连接树莓派和电脑，使其处于同一网段下，然后通过PuTTY配置
IP地址：raspberrypi.local 端口：22
连接成功后输入登录名pi和密码raspberry,接下来简单配置并连接与电脑相同的WiFi；
可下载一个VNC Viewer用于连接到树莓派的桌面。

·2 OpenCV安装
安装并编译OpenCV的资源库
注意接下来都要在cv的环境中操作

硬件的连接

主要器材：
航模电池：给Arduino和电机驱动模块供电；
降压模块：用于降压保护Arduino和电机，并可以稳定电压；
电机驱动模块（L298N）：给电机供电，左侧两个电机的正极同时连接电机驱动模块上的一个out3，负极连接到out4，右侧连接out1、out2；
超声波模块：直接接在Arduino板块上，由Arduino供电并且由Arduino读取返回的信号值，即反馈小车当前的位置状态。

Step1. 将树莓派与Arduino通过USB线（通用串行总线）建立连接。【Arduino在树莓派中是靠串口连接通信的，有两种：USB连接和GPIO通信(其中树莓派的RX引脚——Arduino的TX引脚，树莓派的TX引脚——Arduino的RX引脚，树莓派的GND引脚——Arduino的GND引脚)】
Step2. 在树莓派终端输入ls /dev/tty*查看两者连接端口的名字，若是出现ttyACM0则二者成功建立通信。
Step3. 编写树莓派与Arduino之间通信的代码

树莓派端：

import serial    
# /dev/ttyACM0 是树莓派上编号为0的USB口（可以在/dev目录下通过观察拔插对应的USB口找到对应的编号）
ser = serial.Serial('/dev/ttyACM0', 9600,timeout=1);   # PL011串口[/dev/ttyACM0]（外部晶振驱动）较 mini串口[/dev/ttyS0]（自带晶振驱动）稳定性高
try:
  while 1:
    ser.write('s');  # 树莓派通过串口发送字符's'
    response = ser.readall();  # 读取串口的返回值
    print response;
except:
  ser.close();  # 抛出异常后关闭

Arduino端：【注意：将程序烧到Arduino后一定要把USB拔掉插到树莓派上，不然会导致串口占用！！！】

void setup() 
{
  Serial.begin(9600); // 波特率9600 bps
}
void loop()
{
  if ( Serial.available())  # 判断串口缓存区是否有数据
    {
      if('s' == Serial.read())
        Serial.println("Hello Raspberry,I am Arduino.");  # Arduino收到字符's'后，就向串口打印"Hello Raspberry,I am Arduino."
     }
}

图像处理

RGB模型：R-红色、G-绿色、B-蓝色；(0,0,0)表示黑色、(255, 255, 255)表示白色。
HSV模型：色调（H:hue）—— 0°(以度为计量单位) ~ 360°(0~2π)，从逆时针开始计算，红色为0°，绿色为120°，蓝色为240°
饱和度（S:saturation）—— 0.0 ~ 1.0
亮度（V:value）—— 0.0(黑色)～1.0(白色)

So, 为什么要采用HSV空间的色域？

RGB/HSV Color Demo

注意：在HSV中，V受光照的影响最大
处理方式：对V进行直方图均衡化处理，使各个亮度值的点数相近。

Step1. 获得红色掩膜（进行按位或运算1|1=1，1|0=0，0|1=0，0|0=0）
获得绿色掩膜
Step2. 中值滤波处理（平滑图像，滤去噪声,保护图像的边缘）
Step3. 寻找轮廓并计算质心 【cv2.findContours()】
比较轮廓重心(center_x, center_y)和外接矩形的重心(x+w//2, y+h//2),
if ((x + w // 2) > center_x):
向左转
else :
向右转

上下位机之间的通信

Step1. 遇障—Arduino向树莓派传达Ready信号，树莓派识别完—树莓派向Arduino传达指令
Step2. 先由树莓派发起通信，Arduino进行回应

上位机代码

上位机
# -*- coding: UTF-8 -*-
'''首先由树莓派发起通讯，告知Arduino可以开始运动了；
Arduino接收到树莓派的信号后开始运动，进入loop循环；
遇到障碍物停车，发送信号，等待树莓派的指示，不进入loop；
树莓派拍照，识别，传送信号给Arduino，Arduino收到信号，开始运动。              
树莓派循环等待信号
'''
import time
import numpy as np
import cv2
import serial

ser = serial.Serial('/dev/ttyACM0', 9600,timeout=1);
font= cv2.FONT_HERSHEY_SIMPLEX
lower_red1=np.array([0,128,40])  # 浅红色阈值下界
higher_red1=np.array([5,255,255])  # 浅红色阈值上界
lower_red2=np.array([156,128,40])  
higher_red2=np.array([180,255,255])  
lower_green=np.array([35,100,46])  # 绿色阈值下界
higher_green=np.array([77,255,255])  # 绿色阈值上界

def cal(cnt):
    M = cv2.moments(cnt)  # 计算第一条轮廓的各阶矩,字典形式
    center_x = int(M["m10"] / (M["m00"]+1e-20))
    center_y = int(M["m01"] / (M["m00"]+1e-20))
    return M,center_x,center_y

def command(text,color,signal):
    ser.write(signal.encode())
    time.sleep(0.1)
    response=ser.readall()
    time.sleep(0.1);
    print response
    cv2.putText(frame, text, (x, y - 5), font, 1.0, color, 2)
    print text
    flag=1

def draw(cnt,(x,y,w,h),rectcolor,edgecentercolor):
    cv2.rectangle(frame, (x, y), (x + w, y + h), rectcolor, 2)  # 将检测到的颜色框起来
    cv2.circle(frame, (x + w // 2, y + h // 2), 7, edgecentercolor, -1)     #质心
    cv2.drawContours(frame, cnt, -1, edgecentercolor, 2)


ser.write('start') #告诉Arduino可以开始运动，进入loop循环开车
response = ser.readall();#从串口读取一个字符串
print response 

while(1):# 等待Ready信号
    print "in waiting"
    flag=0 #得到命令flag=1,若是没有得到命令flag=0,继续进行摄像头的工作;每次小车停止后flag信息归零
    ser.write('s');#发起通信，等待小车遇到障碍物
    time.sleep(0.1);
    response = ser.readall();#读取遇到障碍物的消息
    time.sleep(0.1);
    print response #Ready
    if (response=='Ready'):
        cap=cv2.VideoCapture(0)#打开摄像头
        if(cap.isOpened()):
            while(flag!=1):#没有得到命令就继续拍照识别
                print 'camera go'
                ret,frame=cap.read()#按帧读取，这是读取一帧
                # 转化为HSV空间
                img_hsv=cv2.cvtColor(frame,cv2.COLOR_BGR2HSV)  
                #cv2.imshow('img_hsv', img_hsv)
                mask_red1=cv2.inRange(img_hsv,lower_red1,higher_red1)
                mask_red2=cv2.inRange(img_hsv,lower_red2,higher_red2)                
                mask_green=cv2.inRange(img_hsv,lower_green,higher_green)
                mask_red=cv2.bitwise_or(mask_red1,mask_red2)

                mask_green = cv2.medianBlur(mask_green, 7)  
                mask_red = cv2.medianBlur(mask_red, 7)  
                mask=cv2.bitwise_or(mask_green,mask_red)  # 按位运算

                # 第一个参数是输入图像，第二个是轮廓检索模式，第三个是轮廓近似方法
                # 对应的返回值分别是图像，轮廓（是一个Python列表，用于存储图像中的所有轮廓），（轮廓的）层析结构
                # RETR_LIST用于提取所有的轮廓，且属于同一级组织轮廓；RETR_EXTERNAL返回最外边的轮廓，忽略子轮廓；RETR_CCOMP返回所有的轮廓并将轮廓分为两级组织结构;RETR_TREE返回所有轮廓，并且创建一个完整的组织结构列表。
                frame1,cnts1, contours1= cv2.findContours(mask_red, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE) # 第三个参数选定的意义在于将轮廓上的冗余点都去掉，压缩轮廓，从而节省内存开支。
                frame3,cnts3, contours3= cv2.findContours(mask_green,cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)

                for cnt in cnts1:
                    M,center_x,center_y=cal(cnt)
                    (x,y,w,h)=cv2.boundingRect(cnt)  # 返回矩阵的四个点，计算轮廓的垂直边界最小矩形，矩形是与图像上下边界平行的
                    if(w>100 and h>100):
                        draw(cnts1,(x,y,w,h),(240, 65, 85),(0, 255, 0))
                        if ((x + w / 2) > center_x):
                            command('LeftRedArrow',(0,0,255),'o') # 由于是红色箭头，所以发出停止的命令
                            flag=1
                        else:
                            command('RightRedArrow',(0,0,255),'p')
                            flag=1

                for cnt in cnts3:
                    M,center_x,center_y=cal(cnt)
                    (x, y, w, h) = cv2.boundingRect(cnt)  
                    if(w>100 and h>100):
                        draw(cnts3,(x,y,w,h),(0, 255, 255),(0, 255, 0))
                        if ((x + w / 2) > center_x):
                            command('LeftGreenArrow',(0,255,0),'m')  
                            flag=1
                        else:                                                          
                            command('RightGreenArrow',(0,255,0),'n')
                            flag=1
                cv2.imshow('frame',frame)
                cv2.waitKey(800)
                ser.flush()
                #if c==27:
                #    break
                cv2.destroyAllWindows()
                if flag==1:
                    print "got instruction"
                    break;
                else :
                    print "did not get instruction"
        cap.release()

下位机代码

#define PIN_PWML 9
#define PIN_PWMR 10
#define PIN_AIN3 5//对应板子上INPUT3左侧轮子
#define PIN_AIN4 4//对应板子上INPUT4
#define PIN_AIN1 2//对应板子上INPUT1右侧
#define PIN_AIN2 3//对应板子上INPUT2
#define Trig 6 //定义超声波信号发出接口
#define Echo 7 //定义超声波信号接收接口
float cm; //距离变量
float temp;  
int flag;
void setup()
{ 
  Serial.begin(9600);
   //测速引脚初始化
  while(Serial.available() && Serial.read()!="start")
  {
    delay(100);
  }

  Serial.print("wheels on");
    //串口初始化

  pinMode(PIN_PWML,OUTPUT);
  pinMode(PIN_PWMR,OUTPUT);
  pinMode(PIN_AIN3,OUTPUT);
  pinMode(PIN_AIN4,OUTPUT);  
  pinMode(PIN_AIN1,OUTPUT);
  pinMode(PIN_AIN2,OUTPUT);
  //超声波控制引脚初始化
  pinMode(Trig, OUTPUT);
  pinMode(Echo, INPUT);
}
void Stop(int t)
{
  digitalWrite(PIN_AIN3,LOW);
  digitalWrite(PIN_AIN4,LOW);
  digitalWrite(PIN_AIN1,LOW);
  digitalWrite(PIN_AIN2,LOW);
  delay(t);
}
void Go() //直线行驶
{
  analogWrite(PIN_PWML,100);
  digitalWrite(PIN_AIN3,HIGH);
  digitalWrite(PIN_AIN4,LOW);
  analogWrite(PIN_PWMR,100);
  digitalWrite(PIN_AIN1,HIGH);
  digitalWrite(PIN_AIN2,LOW);

}
void Right(int t)  //右转
{
  analogWrite(PIN_PWML,250);
  digitalWrite(PIN_AIN3,HIGH);
  digitalWrite(PIN_AIN4,LOW);
  analogWrite(PIN_PWMR,100);
  digitalWrite(PIN_AIN1,LOW);
  digitalWrite(PIN_AIN2,HIGH);
  delay(t);
}
void Left(int t)
{
  analogWrite(PIN_PWML,7);//
  digitalWrite(PIN_AIN3,HIGH);
  digitalWrite(PIN_AIN4,LOW);
  analogWrite(PIN_PWMR,180);
  digitalWrite(PIN_AIN1,HIGH);
  digitalWrite(PIN_AIN2,LOW);
  delay(t);
}

void Wave()
{
   //给Trig发送一个低高低的短时间脉冲,触发测距
  digitalWrite(Trig, LOW); //给Trig发送一个低电平
  delayMicroseconds(2);    //等待 2微妙
  digitalWrite(Trig,HIGH); //给Trig发送一个高电平
  delayMicroseconds(10);    //等待 10微妙
  digitalWrite(Trig, LOW); //给Trig发送一个低电平
  
  temp = float(pulseIn(Echo, HIGH)); //存储回波等待时间,
  cm = (temp * 17 )/1000; //把回波时间换算成cm
  //声速是340m/1s，换算成34000cm/1000000μs=34/1000 
 //距离(厘米)=(回波时间*(34/1000))/2=回波时间*17/1000
 // Serial.print("Echo =");
 // Serial.print(temp);//串口输出等待时间的原始数据
  //Serial.print(" | | Distance = ");
  //Serial.print(cm);//串口输出距离换算成cm的结果
  //Serial.println("cm");
  //delay(100);
}
void loop() {
    Wave();
    if(cm<=40)
    {
        Stop(100);
        flag=0;
         if(Serial.available()  &&  's'==Serial.read())//一个s对应一个ready对应一次停止
        {
            
            Serial.print("Ready");//停车并且就绪准备读取图片信息
            while(Serial.read() && flag!=1)//读取串口信息
            {
              delay(100); 
              char temp=Serial.read(); 
                  
              switch(temp)//判断指令
              {
               case 'm':{Serial.print("L");flag=1;Left(850);Stop(500);break;}
               case 'n':{Serial.print("R");flag=1;Right(700);Stop(500);break;}
               case 'o':{Serial.print("1");flag=1;Stop(1000);break;}
               case 'p':{Serial.print("2");flag=1;Stop(1000);break;}
               }
               //Serial.flush();
            }
       }     
        Wave();//继续进行超声波检测
    }
    else
      Go();
}
